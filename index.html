        function calculateDistribution() {

            const amount = parseFloat(document.getElementById("amount").value) || 0;

            const categories = [

                { name: "Soziale Wohlfahrt", percentage: 35.10 },

                { name: "Finanzen und Steuern", percentage: 16.10 },

                { name: "Verkehr", percentage: 12.50 },

                { name: "Bildung und Forschung", percentage: 9.70 },

                { name: "Sicherheit", percentage: 8.20 },

                { name: "Beziehungen zum Ausland", percentage: 4.40 },

                { name: "Landwirtschaft und Ernährung", percentage: 4.20 },

                { name: "Übrige Aufgaben", percentage: 9.80 }

            ];

            

            let tableBody = "";

            categories.forEach(category => {

                let value = (amount * category.percentage / 100).toFixed(2);

                tableBody += `<tr><td>${category.name}</td><td>${value} CHF</td></tr>`;

            });

            

            document.getElementById("results").innerHTML = tableBody;

        }

    </script>
